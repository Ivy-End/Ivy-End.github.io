#include <iostream>
 
using namespace std;
 
const int pDay[] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
 
int main()
{
    int N, M, ans = 0;
    cin >> N >> M;
    if((M > 12 || N > 31) ||
        (M == 4 || M == 6 || M == 9 || M == 11) && N > 30 ||
        M == 2 && N > 28)
    {
        cout << "Impossible" << endl;
    }
    else
    {
        for(int i = 1; i < M; i++)
        { ans += pDay[i]; }
        ans += N;
        cout << (ans % 7 == 0 ? 7 : ans % 7) << endl;
    }
    return 0;
}
