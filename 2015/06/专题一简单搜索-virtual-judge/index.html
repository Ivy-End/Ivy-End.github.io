int Solve();
int IsFlip(int x, int y);

{
    while(cin >> M >> N)
    {
        int ans = 2147483647;
        for(int i = 1; i <= M; i++)
        {
            for(int j = 1; j <= N; j++)
            { cin >> pMap[i][j]; }

        {
            memset(pTmp, 0, sizeof(pTmp));

                if(i & (1 << j - 1))
                { pTmp[1][j] = 1; }
            }
            int nTmp = Solve();
            if(nTmp == -1) { continue; }
            if(nTmp < ans)
            {
                ans = nTmp;

        {
            for(int i = 1; i <= M; i++)
            {
                for(int j = 1; j <= N; j++)

}

{
    int nRet = 0;
    for(int i = 2; i <= M; i++)
    {
        for(int j = 1; j <= N; j++)
        {
            if(IsFlip(i - 1, j)) { pTmp[i][j] = 1; }
        }
    }
    for(int i = 1; i <= N; i++)
    {
        if(IsFlip(M, i)) { return -1; }
    }
    for(int i = 1; i <= M; i++)
    {
        for(int j = 1; j <= N; j++)
        {
            nRet += pTmp[i][j];
        }
    }
    return nRet;
}
 
int IsFlip(int x, int y)
{

    for(int i = 0; i < 4; i++)

        int ny = y + dy[i];

        { nRet += pTmp[nx][ny]; }
    }

    return nRet & 1;
}
