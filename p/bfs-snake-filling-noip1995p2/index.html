#include <iostream>
#include <memory.h>

using namespace std;

const int MAX = 16;
const int dx[] = { 1, 0, -1, 0 };
const int dy[] = { 0, -1, 0, 1 };

int x, y, i, nNum = 1;
int N, f[MAX][MAX];

int main()
{
    memset(f, 0, sizeof(f));
    cin >> N;
    x = 1; y = N; i = 0;
    f[x][y] = nNum++;
    while(nNum <= N * N)
    {
        x += dx[i]; y += dy[i];
        if(x >= 1 && x <= N && y >= 1 && y <= N && f[x][y] == 0)
        { f[x][y] = nNum++; }
        else
        {
            x -= dx[i]; y -= dy[i];
            i = (i + 1) % 4;
        }
    }
    for(int i = 1; i <= N; i++)
    {
        for(int j = 1; j <= N; j++)
        {
            if(N * N > 99 && f[i][j] < 100) { cout << " "; }
            if(N * N > 9 && f[i][j] < 10) { cout << " "; }
            cout << f[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
