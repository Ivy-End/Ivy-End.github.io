            private float startX, startY;   // èµ·å§‹ä½ç½®
            private float endX, endY;   // ç»ˆäº†ä½ç½®
            private float offsetX, offsetY; // åç§»é‡
            @Override
            public boolean onTouch(View v, MotionEvent event) {
                
                switch(event.getAction()) {
                    case MotionEvent.ACTION_DOWN:
                        startX = event.getX();
                        startY = event.getY();
                        break;
                    case MotionEvent.ACTION_UP:
                        endX = event.getX();
                        endY = event.getY();

                        offsetX = endX - startX;
                        offsetY = endY - startY;

                        if(Math.abs(offsetX) > Math.abs(offsetY)) { // æ°´å¹³
                            if(offsetX < -5) {
                                System.out.println("Left");

                                System.out.println("Right");
                            }
                        } else { // å‚ç›´
                            if(offsetX < -5) {
                                System.out.println("Up");
                            } else if(offsetX > 5) {
                                System.out.println("Down");
                            }
                        }
                }
                return false;
            }
        });
    }
